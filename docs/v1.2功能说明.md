```python
class DependencyPropagator:
    """资源依赖传播引擎"""
    
    def __init__(self, db: KnowledgeDB):
        self.dependency_graph = self._build_graph()
    
    def _build_graph(self) -> DiGraph:
        """从数据库加载依赖关系，构建有向图"""
        
    def propagate_alert(
        self, 
        alert: Alert, 
        max_depth: int = 3
    ) -> list[PropagatedAlert]:
        """
        传播告警到依赖的下游资源
        
        例如：CPU 告警 → 传播到 数据库会话 → 传播到 业务请求
        """
        
    def find_root_cause(
        self, 
        symptom_alert: Alert
    ) -> RootCauseAnalysis:
        """
        逆向分析：从症状告警追溯可能的根因
        """
        
    def get_upstream_resources(self, resource_id: str, max_depth: int = 3) -> list[str]:
        """获取上游资源列表"""
        
    def get_downstream_resources(self, resource_id: str, max_depth: int = 3) -> list[str]:
        """获取下游资源列表"""
```

#### 2.4.3 混合检索

```python
class HybridSearch:
    """混合检索（关键词 + 向量）"""
    
    def __init__(self, kb: KnowledgeBaseService):
        self.kb = kb
    
    def search(
        self,
        query: str,
        entity_type: str,  # 'alert_rule' / 'sop' / 'case'
        filters: SearchFilters = None,
        keyword_weight: float = 0.3,
        vector_weight: float = 0.7,
        limit: int = 10
    ) -> list[SearchResult]:
        """
        混合检索：
        1. 关键词检索（BM25）
        2. 向量检索（语义相似度）
        3. 加权融合
        """
        
        # 1. 关键词检索
        keyword_results = self.kb.keyword_search(query, entity_type, filters, limit*2)
        
        # 2. 向量检索
        query_embedding = self.kb.embedding_service.embed(query)
        vector_results = self.kb.vector_index.search(query_embedding, top_k=limit*2)
        
        # 3. 加权融合
        fused = self._fuse_results(keyword_results, vector_results, keyword_weight, vector_weight)
        
        return fused[:limit]
    
    def _fuse_results(
        self,
        keyword_results: list,
        vector_results: list,
        kw_weight: float,
        vec_weight: float
    ) -> list:
        # RRF (Reciprocal Rank Fusion) 算法
        k = 60  # RRF 参数
        scores = defaultdict(float)
        
        for rank, result in enumerate(keyword_results):
            scores[result.id] += kw_weight * (1 / (k + rank + 1))
        
        for rank, result in enumerate(vector_results):
            scores[result.id] += vec_weight * (1 / (k + rank + 1))
        
        return sorted(scores.items(), key=lambda x: x[1], reverse=True)
```

#### 2.5.2 覆盖度评估流程

```python
async def evaluate_coverage(self, time_range: TimeRange) -> CoverageReport:
    """评估知识库覆盖度"""
    
    # Step 1: 采集可观测数据
    alerts = await self.collector.collect_alerts(time_range)
    metrics = await self.collector.collect_metrics(time_range)
    logs = await self.collector.collect_logs(time_range, level="error")
    
    # Step 2: 提取告警类型
    alert_types = set(a.alert_type for a in alerts)
    
    # Step 3: 查询已有知识覆盖
    existing_rules = self.kb.list_alert_rules()
    covered_alert_types = set(r.alert_type for r in existing_rules)
    
    # Step 4: LLM 分析未覆盖告警
    uncovered = alert_types - covered_alert_types
    
    prompt = f"""
    你是知识库评估专家。
    
    近期未覆盖的告警类型：{list(uncovered)}
    告警详情：{self._format_alerts(alerts)}
    错误日志样例：{logs[:5]}
    
    分析：
    1. 为每个未覆盖告警分析可能的根因
    2. 评估是否需要生成告警规则
    3. 评估是否需要生成 SOP
    4. 评估是否需要更新观察点元数据
    
    输出 JSON：
    {{
        "uncovered_analysis": [
            {{
                "alert_type": "XXX",
                "estimated_root_causes": ["原因1", "原因2"],
                "needs_rule": true/false,
                "needs_sop": true/false,
                "needs_observation_point": true/false,
                "priority": "high/medium/low"
            }}
        ],
        "knowledge_gaps": ["gap1", "gap2"],
        "recommendations": ["建议1", "建议2"]
    }}
    """
    
    response = await self.llm.generate(prompt, schema=CoverageReportSchema)
    return CoverageReport(**response)
```

```python
async def generate_knowledge(
    self,
    uncovered_alert: UncoveredAlert,
    context: dict
) -> GeneratedKnowledge:

    # 构造生成 prompt

    相关 SOP：{context.get('related_sops', [])}
    
    请生成：
    1. 告警规则（如 needs_rule=True）
    2. SOP（如 needs_sop=True）

    """
    
    response = await self.llm.generate(prompt)
    
    # 解析生成结果
    generated = GeneratedKnowledge.parse_yaml(response)
    
    # 添加元数据
    generated.metadata = {
        "generated_by": "llm",
        "source_alert": uncovered_alert.alert_type,
        "confidence": uncovered_alert.priority,
        "created_at": datetime.now().isoformat(),
    }
```

#### 2.5.4 验证流程

```python
def validate_knowledge(self, knowledge: KnowledgeEntry) -> ValidationResult:
    """验证知识条目"""
    
    errors = []
    warnings = []
    
    # Schema 校验
    schema_errors = self._validate_schema(knowledge)
    errors.extend(schema_errors)
    
    # 重复检测（向量相似度）
    duplicates = self._find_duplicates(knowledge)
    if duplicates:
        warnings.append(f"发现相似知识条目：{[d.id for d in duplicates]}")
        if any(d.similarity > 0.9 for d in duplicates):
            errors.append("与现有条目高度重复（相似度>0.9），请合并")
    
    # 风险评估
    risk_level = self._assess_risk(knowledge)
    
    return ValidationResult(
        valid=len(errors) == 0,
        errors=errors,
        warnings=warnings,
        risk_level=risk_level,
        requires_approval=risk_level in ["L4", "L5"]
    )
```
