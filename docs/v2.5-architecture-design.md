# Javis-DB-Agent V2.5 架构设计文档
## 本地数据库自动发现与纳管

> 版本: 2.5.0 | 日期: 2026-03-31 | 架构师: 道衍

## 一、需求概述

### 1.1 目标
实现**本地主机数据库自动发现、识别与纳管**，不依赖zCloud平台，使Javis-DB-Agent能够：
- 自动扫描本地主机上的数据库实例（PostgreSQL、MySQL等）
- 自动识别数据库类型、版本、端口、运行状态
- 将发现的实例自动注册到本地SQLite注册表
- 建立ChromaDB知识库存储schema、配置、故障案例
- 通过本地Ollama qwen3.5:35b实现自然语言运维查询

### 1.2 核心价值
- **零配置接入**：无需手动配置，发现即纳管
- **统一运维入口**：多类型数据库统一管理
- **本地LLM增强**：离线可用，语义理解能力
- **知识沉淀**：将数据库知识向量化，支持语义检索

```python
# src/discovery/scanner.py
"""
本地数据库扫描器
支持端口扫描 + 进程扫描 双通道发现
"""

import socket
import psutil
import asyncio
import re
from dataclasses import dataclass, field
from typing import List, Optional, Callable
from enum import Enum


class DBType(Enum):
    POSTGRESQL = "postgresql"
    MYSQL = "mysql"
    ORACLE = "oracle"
    MARIADB = "mariadb"
    UNKNOWN = "unknown"


@dataclass
class DiscoveredInstance:
    """发现的数据库实例"""
    db_type: DBType
    host: str = "localhost"
    port: int = 0
    version: str = ""
    process_name: str = ""
    pid: Optional[int] = None
    status: str = "unknown"  # unknown / reachable / authenticated
    process_path: str = ""
    socket_family: str = ""  # ipv4 / ipv6
    
    @property
    def instance_id(self) -> str:
        return f"{self.db_type.value}:{self.host}:{self.port}"


class DatabaseScanner:
    """
    数据库扫描器
    
    发现策略：
    1. 端口扫描：检测常见数据库端口（5432, 3306, 1521等）
    2. 进程扫描：通过psutil扫描运行中的数据库进程
    3. 连接测试：验证可连接性（无需认证的info查询）
    """
    
    # 常见数据库端口
    DEFAULT_DB_PORTS = {
        5432: DBType.POSTGRESQL,
        3306: DBType.MYSQL,
        1521: DBType.ORACLE,
        27017: DBType.UNKNOWN,  # MongoDB (预留)
    }
    
    # 进程名匹配模式
    PROCESS_PATTERNS = {
        DBType.POSTGRESQL: ["postgres", "postmaster", "pg"],
        DBType.MYSQL: ["mysqld", "mariadb"],
        DBType.ORACLE: ["oracle", "tnslsnr"],
    }
    
    # 扫描超时(秒)
    SCAN_TIMEOUT = 2.0
    
    def __init__(
        self,
        ports: Optional[List[int]] = None,
        scan_timeout: float = SCAN_TIMEOUT,
        max_concurrent: int = 50,
    ):
        self.ports = ports or list(self.DEFAULT_DB_PORTS.keys())
        self.scan_timeout = scan_timeout
        self.max_concurrent = max_concurrent
    
    async def scan_ports(self, host: str = "localhost") -> List[DiscoveredInstance]:
        """异步端口扫描，检测常见数据库端口是否开放"""
        results = []
        sem = asyncio.Semaphore(self.max_concurrent)
        
        async def check_port(port: int) -> Optional[DiscoveredInstance]:
            async with sem:
                db_type = self.DEFAULT_DB_PORTS.get(port, DBType.UNKNOWN)
                if await self._is_port_open(host, port):
                    return DiscoveredInstance(
                        db_type=db_type,
                        host=host,
                        port=port,
                        status="reachable" if db_type != DBType.UNKNOWN else "unknown",
                    )
                return None
        
        tasks = [check_port(p) for p in self.ports]
        discovered = await asyncio.gather(*tasks)
        results = [d for d in discovered if d is not None]
        return results
    
    async def _is_port_open(self, host: str, port: int) -> bool:
        """检测端口是否开放"""
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(self.scan_timeout)
            result = sock.connect_ex((host, port))
            sock.close()
            return result == 0
        except (socket.error, OSError):
            return False
    
    def scan_processes(self) -> List[DiscoveredInstance]:
        """进程扫描，通过psutil查找运行中的数据库进程"""
        results = []
        for proc in psutil.process_iter(["pid", "name", "exe", "cmdline"]):
            try:
                name = proc.info["name"] or ""
                exe = proc.info["exe"] or ""
                
                for db_type, patterns in self.PROCESS_PATTERNS.items():
                    if any(p.lower() in name.lower() for p in patterns):
                        instance = self._proc_to_instance(proc, db_type)
                        if instance:
                            results.append(instance)
                            break
            except (psutil.NoSuchProcess, psutil.AccessDenied):
                continue
        return results
    
    def _proc_to_instance(
        self, proc: psutil.Process, db_type: DBType
    ) -> Optional[DiscoveredInstance]:
        """从进程信息构造DiscoveredInstance"""
        try:
            connections = proc.connections(kind="inet")
            ports = set()
            for conn in connections:
                if conn.status == "LISTEN":
                    ports.add(conn.laddr.port)
            
            # 优先使用标准端口
            default_port = next(
                (p for p, t in self.DEFAULT_DB_PORTS.items() if t == db_type),
                0
            )
            port = next((p for p in ports if p == default_port), next(iter(ports), 0))
            
            if not port:
                return None
            
            return DiscoveredInstance(
                db_type=db_type,
                host="localhost",
                port=port,
                pid=proc.pid,
                process_name=proc.name(),
                process_path=proc.exe() or "",
                status="process_found",
            )
        except (psutil.NoSuchProcess, psutil.AccessDenied, psutil.ZombieProcess):
            return None
    
    async def scan_local(self) -> List[DiscoveredInstance]:
        """
        综合扫描：端口扫描 + 进程扫描
        合并结果，按(db_type, port)去重
        """
        port_results, proc_results = await asyncio.gather(
            self.scan_ports("localhost"),
            asyncio.to_thread(self.scan_processes),
        )
        
        seen = {}
        for instance in (*port_results, *proc_results):
            key = (instance.db_type, instance.port)
            if key not in seen:
                if instance.pid and not seen.get(key, DiscoveredInstance(DBType.UNKNOWN)).pid:
                    seen[key] = instance
                elif not instance.pid:
                    seen[key] = instance
        
        return list(seen.values())
```

```python
# src/discovery/identifier.py
"""
数据库类型识别与版本检测
"""

import asyncio
import re
from typing import Optional
from dataclasses import dataclass

try:
    import asyncpg
except ImportError:
    asyncpg = None

try:
    import aiomysql
except ImportError:
    aiomysql = None

from .scanner import DBType, DiscoveredInstance


@dataclass
class IdentifiedInstance:
    """识别后的完整实例信息"""
    instance: DiscoveredInstance
    version: str
    version_major: int
    version_minor: int
    edition: str = ""  # MySQL Community / Enterprise, PG Standard / Enterprise
    max_connections: int = 100
    current_connections: int = 0


class DatabaseIdentifier:
    """
    数据库识别器
    
    功能：
    1. 连接测试（无需认证的info查询）
    2. 版本检测
    3. 类型确认（区分MySQL/MariaDB）
    """
    
    SCAN_TIMEOUT = 3.0
    MARIADB_INDICATORS = ["MariaDB", "maria", "mariadb"]
    
    async def identify(self, instance: DiscoveredInstance) -> Optional[IdentifiedInstance]:
        """识别单个实例，返回完整信息，失败返回None"""
        if instance.db_type == DBType.POSTGRESQL:
            return await self._identify_postgres(instance)
        elif instance.db_type in (DBType.MYSQL, DBType.MARIADB):
            return await self._identify_mysql(instance)
        elif instance.db_type == DBType.ORACLE:
            return await self._identify_oracle(instance)
        return None
    
    async def identify_all(
        self, instances: list[DiscoveredInstance]
    ) -> list[IdentifiedInstance]:
        """批量识别"""
        results = await asyncio.gather(
            *[self.identify(inst) for inst in instances],
            return_exceptions=True,
        )
        return [r for r in results if isinstance(r, IdentifiedInstance)]
    
    async def _identify_postgres(self, instance: DiscoveredInstance) -> Optional[IdentifiedInstance]:
        """识别PostgreSQL实例"""
        if asyncpg is None:
            return None
        try:
            conn = await asyncpg.connect(
                host=instance.host,
                port=instance.port,
                timeout=self.SCAN_TIMEOUT,
                command_timeout=self.SCAN_TIMEOUT,
            )
            try:
                version_row = await conn.fetchrow("SELECT version()")
                version_str = version_row["version"]
                version_parts = version_str.split("(")[0].strip().split()
                version = version_parts[-1] if len(version_parts) >= 2 else version_str
                
                version_match = re.search(r"(\d+)\.(\d+)", version)
                major = int(version_match.group(1)) if version_match else 0
                minor = int(version_match.group(2)) if version_match else 0
                
                conn_count = await conn.fetchval("SELECT count(*) FROM pg_stat_activity")
                max_conn = await conn.fetchval("SHOW max_connections")
                
                instance.version = version
                instance.status = "identified"
                
                return IdentifiedInstance(
                    instance=instance,
                    version=version,
                    version_major=major,
                    version_minor=minor,
                    max_connections=int(max_conn) if max_conn else 100,
                    current_connections=conn_count or 0,
                )
            finally:
                await conn.close()
        except Exception:
            return None
    
    async def _identify_mysql(self, instance: DiscoveredInstance) -> Optional[IdentifiedInstance]:
        """识别MySQL/MariaDB实例"""
        if aiomysql is None:
            return None
        try:
            conn = await aiomysql.connect(
                host=instance.host,
                port=instance.port,
                timeout=self.SCAN_TIMEOUT,
            )
            async with conn.cursor(aiomysql.DictCursor) as cursor:
                await cursor.execute("SELECT VERSION() as version, @@max_connections as max_conn")
                row = await cursor.fetchone()
                
                version_str = row["version"]
                is_mariadb = any(
                    indicator in version_str.lower() 
                    for indicator in self.MARIADB_INDICATORS
                )
                edition = "MariaDB" if is_mariadb else "MySQL"
                
                version_match = re.search(r"(\d+)\.(\d+)", version_str)
                major = int(version_match.group(1)) if version_match else 0
                minor = int(version_match.group(2)) if version_match else 0
                
                if is_mariadb:
                    instance.db_type = DBType.MARIADB
                
                instance.version = version_str
                instance.status = "identified"
                
                return IdentifiedInstance(
                    instance=instance,
                    version=version_str,
                    version_major=major,
                    version_minor=minor,
                    edition=edition,
                    max_connections=int(row["max_connections"]) if row["max_connections"] else 100,
                )
        except Exception:
            return None
    
    async def _identify_oracle(self, instance: DiscoveredInstance) -> Optional[IdentifiedInstance]:
        """识别Oracle实例（预留，当前仅端口确认）"""
        instance.status = "unverified_requires_oracle_client"
        return IdentifiedInstance(
            instance=instance,
            version="unknown",
            version_major=0,
            version_minor=0,
        )
```

```python
# src/discovery/registry.py
"""
本地SQLite注册表 - 纳管实例存储
"""

import sqlite3
import uuid
from datetime import datetime
from dataclasses import dataclass, asdict
from typing import List, Optional
from contextlib import contextmanager
from pathlib import Path

from .scanner import DBType
from .identifier import IdentifiedInstance


@dataclass
class ManagedInstance:
    """已纳管的数据库实例"""
    id: str
    db_type: str
    host: str
    port: int
    version: str
    version_major: int
    version_minor: int
    edition: str
    status: str  # discovered / onboarded / monitoring / error / removed
    discovered_at: str  # ISO8601
    onboarded_at: Optional[str] = None
    last_check_at: Optional[str] = None
    max_connections: int = 100
    current_connections: int = 0
    process_path: str = ""
    pid: Optional[int] = None
    metadata_: str = "{}"  # JSON扩展字段
    
    @classmethod
    def from_identified(
        cls, identified: IdentifiedInstance, status: str = "discovered"
    ) -> "ManagedInstance":
        inst = identified.instance
        now = datetime.utcnow().isoformat()
        return cls(
            id=str(uuid.uuid4()),
            db_type=inst.db_type.value,
            host=inst.host,
            port=inst.port,
            version=identified.version,
            version_major=identified.version_major,
            version_minor=identified.version_minor,
            edition=identified.edition,
            status=status,
            discovered_at=now,
            onboarded_at=now if status == "onboarded" else None,
            last_check_at=now,
            max_connections=identified.max_connections,
            current_connections=identified.current_connections,
            process_path=inst.process_path,
            pid=inst.pid,
        )


class LocalRegistry:
    """
    本地SQLite注册表
    
    表结构：
    - managed_instances: 已纳管实例
    - status_history:    状态变更历史
    - scan_sessions:     扫描会话记录
    """
    
    SCHEMA_SQL = """
    CREATE TABLE IF NOT EXISTS managed_instances (
        id              TEXT PRIMARY KEY,
        db_type         TEXT NOT NULL,
        host            TEXT NOT NULL DEFAULT 'localhost',
        port            INTEGER NOT NULL,
        version         TEXT DEFAULT '',
        version_major   INTEGER DEFAULT 0,
        version_minor   INTEGER DEFAULT 0,
        edition         TEXT DEFAULT '',
        status          TEXT NOT NULL DEFAULT 'discovered',
        discovered_at   TEXT NOT NULL,
        onboarded_at    TEXT,
        last_check_at   TEXT,
        max_connections INTEGER DEFAULT 100,
        current_connections INTEGER DEFAULT 0,
        process_path    TEXT DEFAULT '',
        pid             INTEGER,
        metadata_       TEXT DEFAULT '{}'
    );
    
    CREATE UNIQUE INDEX IF NOT EXISTS idx_instance_key 
        ON managed_instances(db_type, host, port);
    
    CREATE INDEX IF NOT EXISTS idx_status 
        ON managed_instances(status);
    
    CREATE TABLE IF NOT EXISTS status_history (
        id              INTEGER PRIMARY KEY AUTOINCREMENT,
        instance_id     TEXT NOT NULL,
        old_status      TEXT,
        new_status      TEXT NOT NULL,
        changed_at      TEXT NOT NULL,
        reason          TEXT DEFAULT '',
        FOREIGN KEY (instance_id) REFERENCES managed_instances(id)
    );
    
    CREATE TABLE IF NOT EXISTS scan_sessions (
        id              TEXT PRIMARY KEY,
        started_at      TEXT NOT NULL,
        finished_at     TEXT,
        instances_found INTEGER DEFAULT 0,
        instances_new   INTEGER DEFAULT 0,
        instances_changed INTEGER DEFAULT 0,
        error_message   TEXT DEFAULT ''
    );
    """
    
    def __init__(self, db_path: str = "data/managed_db.db"):
        self.db_path = Path(db_path)
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        self._init_db()
    
    @contextmanager
    def _conn(self):
        """上下文管理器：自动提交/回滚"""
        conn = sqlite3.connect(str(self.db_path))
        conn.row_factory = sqlite3.Row
        try:
            yield conn
            conn.commit()
        except Exception:
            conn.rollback()
            raise
        finally:
            conn.close()
    
    def _init_db(self):
        """初始化数据库schema"""
        with self._conn() as conn:
            conn.executescript(self.SCHEMA_SQL)
    
    def register(self, instance: ManagedInstance) -> str:
        """注册新实例（upsert），已存在则更新，返回instance_id"""
        with self._conn() as conn:
            conn.execute("""
                INSERT INTO managed_instances (
                    id, db_type, host, port, version, version_major, version_minor,
                    edition, status, discovered_at, onboarded_at, last_check_at,
                    max_connections, current_connections, process_path, pid, metadata_
                ) VALUES (
                    :id, :db_type, :host, :port, :version, :version_major, :version_minor,
                    :edition, :status, :discovered_at, :onboarded_at, :last_check_at,
                    :max_connections, :current_connections, :process_path, :pid, :metadata_
                )
                ON CONFLICT(db_type, host, port) DO UPDATE SET
                    version = excluded.version,
                    version_major = excluded.version_major,
                    version_minor = excluded.version_minor,
                    edition = excluded.edition,
                    status = excluded.status,
                    last_check_at = excluded.last_check_at,
                    max_connections = excluded.max_connections,
                    current_connections = excluded.current_connections,
                    pid = excluded.pid
            """, asdict(instance))
            
            row = conn.execute(
                "SELECT id FROM managed_instances WHERE db_type=? AND host=? AND port=?",
                (instance.db_type, instance.host, instance.port)
            ).fetchone()
            return row["id"]
    
    def get_all(self, status_filter: Optional[str] = None) -> List[ManagedInstance]:
        """查询所有实例"""
        with self._conn() as conn:
            if status_filter:
                rows = conn.execute(
                    "SELECT * FROM managed_instances WHERE status=? ORDER BY discovered_at DESC",
                    (status_filter,)
                ).fetchall()
            else:
                rows = conn.execute(
                    "SELECT * FROM managed_instances ORDER BY discovered_at DESC"
                ).fetchall()
            return [ManagedInstance(**dict(row)) for row in rows]
    
    def get_by_id(self, instance_id: str) -> Optional[ManagedInstance]:
        """按ID查询"""
        with self._conn() as conn:
            row = conn.execute(
                "SELECT * FROM managed_instances WHERE id=?",
                (instance_id,)
            ).fetchone()
            return ManagedInstance(**dict(row)) if row else None
    
    def get_by_key(self, db_type: str, host: str, port: int) -> Optional[ManagedInstance]:
        """按(db_type, host, port)唯一键查询"""
        with self._conn() as conn:
            row = conn.execute(
                "SELECT * FROM managed_instances WHERE db_type=? AND host=? AND port=?",
                (db_type, host, port)
            ).fetchone()
            return ManagedInstance(**dict(row)) if row else None
    
    def update_status(
        self, instance_id: str, new_status: str, reason: str = ""
    ) -> bool:
        """更新实例状态并记录历史"""
        with self._conn() as conn:
            row = conn.execute(
                "SELECT status FROM managed_instances WHERE id=?",
                (instance_id,)
            ).fetchone()
            if not row:
                return False
            
            old_status = row["status"]
            now = datetime.utcnow().isoformat()
            
            conn.execute(
                "UPDATE managed_instances SET status=?, last_check_at=? WHERE id=?",
                (new_status, now, instance_id)
            )
            conn.execute(
                "INSERT INTO status_history (instance_id, old_status, new_status, changed_at, reason) VALUES (?, ?, ?, ?, ?)",
                (instance_id, old_status, new_status, now, reason)
            )
            return True
    
    def update_connections(self, instance_id: str, current: int) -> bool:
        """更新连接数"""
        with self._conn() as conn:
            result = conn.execute(
                "UPDATE managed_instances SET current_connections=?, last_check_at=? WHERE id=?",
                (current, datetime.utcnow().isoformat(), instance_id)
            )
            return result.rowcount > 0
    
    def remove(self, instance_id: str) -> bool:
        """移除实例（软删除，标记为removed）"""
        return self.update_status(instance_id, "removed", "user_requested")
    
    def get_stats(self) -> dict:
        """获取注册统计"""
        with self._conn() as conn:
            total = conn.execute("SELECT COUNT(*) FROM managed_instances").fetchone()[0]
            by_status = {
                row["status"]: row["cnt"]
                for row in conn.execute(
                    "SELECT status, COUNT(*) as cnt FROM managed_instances GROUP BY status"
                )
            }
            by_type = {
                row["db_type"]: row["cnt"]
                for row in conn.execute(
                    "SELECT db_type, COUNT(*) as cnt FROM managed_instances GROUP BY db_type"
                )
            }
            return {
                "total_instances": total,
                "by_status": by_status,
                "by_type": by_type,
            }
```

```python
class LocalKnowledgeBase:
    """
    本地知识库 - 基于ChromaDB
    
    集合：
    - schemas:     数据库Schema知识
    - configs:     配置参数知识
    - cases:       故障案例知识
    """
    
    PERSIST_DIR = "data/local_knowledge"
    
    def __init__(self, persist_dir: Optional[str] = None):
        if not CHROMA_AVAILABLE:
            raise RuntimeError("ChromaDB not installed: pip install chromadb")
        
        self.persist_dir = persist_dir or self.PERSIST_DIR
        self._client = None

        import os
        os.makedirs(self.persist_dir, exist_ok=True)
        self._client = chromadb.PersistentClient(path=self.persist_dir)
        
        for name, desc in [
            ("schemas", "Database schema knowledge"),
            ("configs", "Database configuration knowledge"),
            ("cases", "Failure case knowledge"),
        ]:
            try:
                coll = self._client.get_or_create_collection(
                    name=name,
                    embedding_function=self._embedding_fn,
                    metadata={"description": desc},
                )
            except Exception:
                coll = self._client.get_collection(name)
            self._collections[name] = coll
    
    def add_schema(self, schema: SchemaKnowledge) -> str:
        """添加Schema知识"""
        content = self._format_schema_content(schema)
        doc_id = f"schema_{schema.instance_id}_{schema.db_name}"
        
        self._collections["schemas"].upsert(
            ids=[doc_id],
            documents=[content],
            metadatas=[{
                "instance_id": schema.instance_id,
                "db_name": schema.db_name,
                "table_count": len(schema.tables),
                "captured_at": schema.captured_at,
                "version": schema.version,
            }]
        )
        return doc_id
    
    def search_schemas(
        self, query: str, instance_id: Optional[str] = None, top_k: int = 5
    ) -> List[Dict]:
        """语义搜索Schema"""
        where = {"instance_id": instance_id} if instance_id else None
        results = self._collections["schemas"].query(
            query_texts=[query],
            n_results=top_k,
            where=where,
            include=["documents", "metadatas", "distances"],
        )
        return self._format_search_results(results)

            n_results=top_k,
            where=where,
            include=["documents", "metadatas", "distances"],
        )
        return self._format_search_results(results)
    
    def add_case(self, case: CaseKnowledge) -> str:
        """添加故障案例"""
        content = self._format_case_content(case)
        doc_id = case.id or f"case_{hashlib.md5(case.title.encode()).hexdigest()[:12]}"
        
        self._collections["cases"].upsert(
            ids=[doc_id],
            documents=[content],
            metadatas=[{
                "instance_id": case.instance_id,
                "db_type": case.db_type,
                "db_version": case.db_version,
                "title": case.title,
                "created_at": case.created_at,
            }]
        )
        return doc_id
    
    def search_cases(
        self, query: str, db_type: Optional[str] = None, top_k: int = 5
    ) -> List[Dict]:
        """语义搜索故障案例"""
        where = {"db_type": db_type} if db_type else None
        results = self._collections["cases"].query(
            query_texts=[query],
            n_results=top_k,
            where=where,
            include=["documents", "metadatas", "distances"],
        )
        return self._format_search_results(results)
    
    def _format_schema_content(self, schema: SchemaKnowledge) -> str:
        tables_desc = "\n".join([
            f"表: {t['table_name']}, 行数: {t.get('row_count', 0)}, "
            f"大小: {t.get('size_bytes', 0)} bytes, "
            f"列: {', '.join(c.get('name', '') for c in t.get('columns', []))}"
            for t in schema.tables
        ])
        return f"""数据库: {schema.db_name}
版本: {schema.version}
采集时间: {schema.captured_at}

表结构:
{tables_desc}
"""
    
    def _format_config_content(self, config: ConfigKnowledge) -> str:
        params = "\n".join([f"  {k}: {v}" for k, v in list(config.parameters.items())[:50]])
        return f"""数据库类型: {config.db_type}
版本: {config.version}
采集时间: {config.captured_at}

配置参数:
{params}
"""
    
    def _format_case_content(self, case: CaseKnowledge) -> str:
        return f"""标题: {case.title}
数据库类型: {case.db_type} {case.db_version}
实例: {case.instance_id}

症状:
{case.symptoms}

根因:
{case.root_cause}

解决方案:
{case.solution}
"""
    
    def _format_search_results(self, results: dict) -> List[Dict]:
        items = []
        if results and results.get("ids"):
            for i in range(len(results["ids"][0])):
                items.append({
                    "id": results["ids"][0][i],
                    "content": results["documents"][0][i] if results["documents"] else "",
                    "metadata": results["metadatas"][0][i] if results["metadatas"] else {},
                    "distance": results["distances"][0][i] if results["distances"] else 0.0,

    def get_stats(self) -> Dict[str, int]:
```

```python
    2. identify() → 识别类型和版本
    3. register() → 注册到SQLite
    4. enrich() → 写入ChromaDB知识库

    def __init__(
        self,
        registry: Optional[LocalRegistry] = None,
        knowledge_base: Optional[LocalKnowledgeBase] = None,
        scanner: Optional[DatabaseScanner] = None,
        identifier: Optional[DatabaseIdentifier] = None,

        self.kb = knowledge_base or LocalKnowledgeBase()
        self.scanner = scanner or DatabaseScanner()
        self.identifier = identifier or DatabaseIdentifier()
        self._running = False

    async def discover_and_onboard(
        self,
        auto_onboard: bool = True,
        capture_schema: bool = True,
    ) -> OnboardingResult:
        """
        完整的发现与纳管流程

        Args:
            auto_onboard: 是否自动纳管（注册到SQLite）
            capture_schema: 是否捕获schema知识

        Returns:
            OnboardingResult: 纳管结果详情
        """
        session_id = str(uuid.uuid4())
        started_at = datetime.utcnow().isoformat()
        result = OnboardingResult(session_id=session_id, started_at=started_at)

        try:
            # Step 1: 扫描
            discovered = await self.scanner.scan_local()
            result.discovered = discovered

            # Step 2: 识别
            identified = await self.identifier.identify_all(discovered)
            result.identified = identified

            # Step 3: 注册
            for ident_inst in identified:
                managed = ManagedInstance.from_identified(
                    ident_inst,
                    status="onboarded" if auto_onboard else "discovered"
                )
                instance_id = self.registry.register(managed)
                managed.id = instance_id
                result.registered.append(managed)

                # Step 4: 知识库充实
                if capture_schema and auto_onboard:
                    await self._capture_instance_knowledge(managed, ident_inst)

            # 统计
            result.new_count = len(result.registered)
            result.changed_count = 0  # TODO: 实现变更检测

        except Exception as e:
            result.errors.append(str(e))

        result.finished_at = datetime.utcnow().isoformat()
        return result
```

```python
class NLQueryResult:

class LLMEnricher:
    """
    LLM语义增强器

    功能：
    1. 自然语言意图识别
    2. SQL查询生成
    3. schema语义摘要
    4. 诊断推理增强
    """

    INTENT_PROMPT = """你是一个数据库运维助手。请分析用户的自然语言查询，识别其运维意图。

支持的意图类型：
- query_session: 查询会话列表
- query_lock: 查询锁信息
- analyze_lock: 分析死锁
- check_replication: 检查复制状态
- query_capacity: 查询容量/磁盘
- query_performance: 查询性能指标
- diagnose: 诊断问题

输出格式（JSON）：
{
  "intent": "意图类型",
  "entities": {
    "instance_id": "实例ID（如果提到）",
    "time_range": "时间范围（如果提到）",
    "filters": "其他过滤条件"
  },
  "generated_sql": "生成的SQL查询（如果适用）",
  "summary": "一句话总结",
  "confidence": 0.0-1.0置信度
}

用户查询：{query}
"""

    def __init__(
        self,
        ollama_client: Optional[OllamaClient] = None,
        knowledge_base: Optional[LocalKnowledgeBase] = None,
    ):
        self.ollama = ollama_client or OllamaClient()
        self.kb = knowledge_base or LocalKnowledgeBase()

    async def understand_query(self, query: str) -> NLQueryResult:
        """
        理解自然语言查询
        返回意图、实体、生成的SQL
        """
        try:
            import json
            response = await self.ollama.complete(
                prompt=self.INTENT_PROMPT.format(query=query),
                system="你是一个数据库运维助手，输出JSON格式。"
            )

            # 解析JSON响应
            start = response.find("{")
            end = response.rfind("}") + 1
            if start >= 0 and end > start:
                data = json.loads(response[start:end])
                return NLQueryResult(
                    intent=data.get("intent", "unknown"),
                    entities=data.get("entities", {}),
                    generated_sql=data.get("generated_sql", ""),
                    summary=data.get("summary", ""),
                    confidence=data.get("confidence", 0.0),
                )
        except Exception:
            pass

        return NLQueryResult(
            intent="unknown",
            entities={},
            summary="无法理解查询",
            confidence=0.0,
        )

    async def summarize_schema(self, instance_id: str, query: str) -> str:
        """
        基于自然语言查询schema知识库
        返回相关的schema摘要
        """
        results = self.kb.search_schemas(
            query=query,
            instance_id=instance_id,
            top_k=3,
        )

        if not results:
            return "未找到相关schema知识"

        summary_parts = ["找到以下相关表结构：\n"]
        for r in results:
            summary_parts.append(f"- {r['content'][:500]}")
        return "\n".join(summary_parts)

    async def diagnose_with_context(
        self,
        symptom: str,
        instance: ManagedInstance,
    ) -> str:
        """
        诊断推理：结合知识库和LLM进行诊断
        """
        # 检索相关故障案例
        cases = self.kb.search_cases(
            query=symptom,
            db_type=instance.db_type,
            top_k=3,
        )

        # 构建诊断上下文
        context = f"数据库类型: {instance.db_type} {instance.version}\n"
        context += f"症状: {symptom}\n\n"
        context += "相关故障案例：\n"
        for c in cases:
            context += f"- {c['content'][:300]}\n\n"

        prompt = f"""你是一个资深数据库管理员。请根据以下信息进行诊断推理。

{context}

请分析：
1. 可能的原因
2. 推荐的检查步骤
3. 可能的解决方案

输出格式：
## 可能原因
...

## 推荐检查
...

## 解决方案
...
"""
        try:
            return await self.ollama.complete(prompt=prompt)
        except Exception as e:
            return f"诊断失败: {str(e)}"
```

### 9.3 实例来源统一抽象

#### 9.3.1 InstanceSourceAdapter

```python
# src/discovery/instance_source_adapter.py
"""
实例来源适配器 - 统一zCloud实例与本地实例的访问接口
"""

from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import List, Optional, Dict, Any
from enum import Enum


class InstanceSource(Enum):
    """实例来源"""
    ZCLOUD = "zcloud"      # 来自zCloud平台
    LOCAL = "local"        # 本地纳管


@dataclass
class UnifiedInstance:
    """统一实例模型"""
    instance_id: str
    source: InstanceSource
    db_type: str           # mysql / postgresql
    host: str
    port: int
    name: str = ""         # 显示名称
    version: str = ""
    status: str = "unknown"
    # zCloud特有
    zcloud_instance_id: Optional[str] = None
    # 本地特有
    local_uuid: Optional[str] = None
    discovered_at: Optional[str] = None
    # 连接信息
    username: str = ""
    password: str = ""     # 加密存储
    metadata: Dict[str, Any] = None
    
    def __post_init__(self):
        self.metadata = self.metadata or {}


class InstanceSourceAdapter(ABC):
    """实例源抽象基类"""
    
    @abstractmethod
    async def list_instances(self, status_filter: Optional[str] = None) -> List[UnifiedInstance]:
        """列出实例"""
        pass
    
    @abstractmethod
    async def get_instance(self, instance_id: str) -> Optional[UnifiedInstance]:
        """获取单个实例"""
        pass
    
    @abstractmethod
    async def health_check(self, instance_id: str) -> Dict[str, Any]:
        """健康检查"""
        pass
    
    @abstractmethod
    async def get_connection_info(self, instance_id: str) -> Dict[str, Any]:
        """获取连接信息（host, port, credentials）"""
        pass


class ZCloudInstanceSource(InstanceSourceAdapter):
    """zCloud实例源适配器"""
    
    def __init__(self, javis_client):
        self.client = javis_client  # Javis API客户端
    
    async def list_instances(self, status_filter: Optional[str] = None) -> List[UnifiedInstance]:
        instances = await self.client.list_instances(status=status_filter)
        return [self._to_unified(i) for i in instances]
    
    async def get_instance(self, instance_id: str) -> Optional[UnifiedInstance]:
        inst = await self.client.get_instance(instance_id)
        return self._to_unified(inst) if inst else None
    
    async def health_check(self, instance_id: str) -> Dict[str, Any]:
        return await self.client.health_check(instance_id)
    
    async def get_connection_info(self, instance_id: str) -> Dict[str, Any]:
        inst = await self.client.get_instance(instance_id)
        return {
            "host": inst.get("host"),
            "port": inst.get("port"),
            "username": inst.get("username", ""),
            "password": inst.get("password", ""),
            "db_type": inst.get("db_type"),
        }
    
    def _to_unified(self, inst: dict) -> UnifiedInstance:
        return UnifiedInstance(
            instance_id=inst["instance_id"],
            source=InstanceSource.ZCLOUD,
            db_type=inst.get("db_type", "postgresql"),
            host=inst.get("host", "localhost"),
            port=inst.get("port", 5432),
            name=inst.get("name", ""),
            version=inst.get("version", ""),
            status=inst.get("status", "unknown"),
            zcloud_instance_id=inst["instance_id"],
        )


class LocalInstanceSource(InstanceSourceAdapter):
    """本地实例源适配器"""
    
    def __init__(self, registry: 'LocalRegistry'):
        self.registry = registry
    
    async def list_instances(self, status_filter: Optional[str] = None) -> List[UnifiedInstance]:
        managed = self.registry.get_all(status_filter=status_filter)
        return [self._to_unified(m) for m in managed]
    
    async def get_instance(self, instance_id: str) -> Optional[UnifiedInstance]:
        managed = self.registry.get_by_id(instance_id)
        return self._to_unified(managed) if managed else None
    
    async def health_check(self, instance_id: str) -> Dict[str, Any]:
        managed = self.registry.get_by_id(instance_id)
        if not managed:
            return {"status": "not_found"}
        
        # 执行健康检查
        try:
            from src.db.direct_postgres_connector import DirectPostgresConnector
            from src.db.mysql_adapter import MySQLConnector
            
            if managed.db_type == "postgresql":
                conn = DirectPostgresConnector(
                    host=managed.host,
                    port=managed.port,
                    user=managed.username or None,
                    password=managed.password or None,
                )
            else:
                conn = MySQLConnector(
                    host=managed.host,
                    port=managed.port,
                    user=managed.username or None,
                    password=managed.password or None,
                )
            
            is_healthy = await conn.health_check()
            await conn.close()
            return {
                "status": "healthy" if is_healthy else "unhealthy",
                "instance_id": instance_id,
            }
        except Exception as e:
            return {
                "status": "error",
                "error": str(e),
                "instance_id": instance_id,
            }
    
    async def get_connection_info(self, instance_id: str) -> Dict[str, Any]:
        managed = self.registry.get_by_id(instance_id)
        if not managed:
            return {}
        return {
            "host": managed.host,
            "port": managed.port,
            "username": managed.username or "",
            "password": managed.password or "",
            "db_type": managed.db_type,
        }
    
    def _to_unified(self, managed: 'ManagedInstance') -> UnifiedInstance:
        return UnifiedInstance(
            instance_id=managed.id,
            source=InstanceSource.LOCAL,
            db_type=managed.db_type,
            host=managed.host,
            port=managed.port,
            name=f"{managed.db_type}:{managed.host}:{managed.port}",
            version=managed.version,
            status=managed.status,
            local_uuid=managed.id,
            discovered_at=managed.discovered_at,
            username="",  # 本地实例暂时不需要认证信息
            password="",
        )


class UnifiedInstanceRegistry:
    """
    统一实例注册中心
    
    同时管理zCloud实例和本地实例，
    提供统一的查询接口给Tool/Agent使用。
    """
    
    def __init__(
        self,
        zcloud_client=None,
        local_registry: 'LocalRegistry' = None,
    ):
        self._zcloud = ZCloudInstanceSource(zcloud_client) if zcloud_client else None
        self._local = LocalInstanceSource(local_registry) if local_registry else None
    
    def set_zcloud_client(self, client):
        """设置zCloud客户端"""
        self._zcloud = ZCloudInstanceSource(client)
    
    def set_local_registry(self, registry: 'LocalRegistry'):
        """设置本地注册表"""
        self._local = LocalInstanceSource(registry)
    
    async def list_all_instances(
        self,
        source_filter: Optional[InstanceSource] = None,
        status_filter: Optional[str] = None,
    ) -> List[UnifiedInstance]:
        """列出所有实例（支持来源过滤）"""
        results = []
        
        if source_filter in (None, InstanceSource.ZCLOUD) and self._zcloud:
            zcloud_instances = await self._zcloud.list_instances(status_filter)
            results.extend(zcloud_instances)
        
        if source_filter in (None, InstanceSource.LOCAL) and self._local:
            local_instances = await self._local.list_instances(status_filter)
            results.extend(local_instances)
        
        return results
    
    async def get_instance(self, instance_id: str) -> Optional[UnifiedInstance]:
        """获取实例（自动识别来源）"""
        # 优先查本地
        if self._local:
            inst = await self._local.get_instance(instance_id)
            if inst:
                return inst
        
        # 查zCloud
        if self._zcloud:
            inst = await self._zcloud.get_instance(instance_id)
            if inst:
                return inst
        
        return None
    
    async def get_connection_info(self, instance_id: str) -> Optional[Dict[str, Any]]:
        """获取连接信息"""
        inst = await self.get_instance(instance_id)
        if not inst:
            return None
        
        if inst.source == InstanceSource.LOCAL and self._local:
            return await self._local.get_connection_info(instance_id)
        elif inst.source == InstanceSource.ZCLOUD and self._zcloud:
            return await self._zcloud.get_connection_info(instance_id)
        
        return None
    
    async def health_check_all(self) -> Dict[str, Dict]:
        """健康检查所有实例"""
        results = {}
        
        if self._local:
            local_all = await self._local.list_instances()
            for inst in local_all:
                results[inst.instance_id] = await self._local.health_check(inst.instance_id)
        
        if self._zcloud:
            zcloud_all = await self._zcloud.list_instances()
            for inst in zcloud_all:
                results[inst.instance_id] = await self._zcloud.health_check(inst.instance_id)
        
        return results
```

```python
class PGSessionTool(BaseTool):
    """PostgreSQL会话查询Tool - 支持本地实例"""
    
    definition = ToolDefinition(
        name="pg_session_analysis",
        description="查询PostgreSQL会话",
        category="query",
        risk_level=RiskLevel.L1_READ,
        params=[
            ToolParam(name="instance_id", type="string", description="实例ID"),
            ToolParam(name="limit", type="int", description="限制数量", required=False, default=100),
        ],
    )
    
    async def pre_execute(self, params: dict, context: dict) -> tuple[bool, Optional[str]]:
        """前置检查：支持本地实例"""
        instance_id = params.get("instance_id")
        if not instance_id:
            return False, "缺少instance_id"
        
        # 获取统一注册中心
        from src.api.instance_registry import get_unified_registry
        registry = get_unified_registry()
        
        # 创建执行上下文（自动识别zCloud或本地实例）
        ctx = await ToolExecutionContext.from_instance(instance_id, registry)
        if not ctx:
            return False, f"实例不存在: {instance_id}"
        
        # 将上下文注入params，供execute使用
        params["_context"] = ctx
        
        return True, None
    
    async def execute(self, params: dict, context: dict) -> ToolResult:
        ctx = params["_context"]
        
        if ctx.source == InstanceSource.LOCAL:
            # 本地实例：使用直连接入器
            connector = ctx.db_connector
        else:
            # zCloud实例：使用API
            connector = get_zcloud_connector(ctx.instance_id)
        
        sessions = await connector.get_sessions(params.get("limit", 100))
        
        return ToolResult(
            success=True,
            data={"sessions": sessions, "count": len(sessions)},
        )
```

```python
        # 如果上下文已指定
        if context.get("instance_id"):
            return context["instance_id"]

            elif inst.source == InstanceSource.ZCLOUD:
                if inst.name and inst.name in goal:
                    return inst.instance_id
        
        # 默认返回第一个可用实例
        if all_instances:

        return None
```

### 9.11 关键设计决策

1. **双源并存**：zCloud实例和本地实例同时支持，不替换原有架构
2. **Adapter模式**：通过 `InstanceSourceAdapter` 抽象差异，对上层透明
3. **Tool层统一**：所有Tool的 `pre_execute` 使用统一的 `ToolExecutionContext.from_instance()`
4. **连接器复用**：`DirectPostgresConnector` 和 `MySQLConnector` 同时服务于zCloud和本地实例
5. **知识库共享**：ChromaDB知识库同时服务于zCloud和本地实例（通过instance_id隔离）
