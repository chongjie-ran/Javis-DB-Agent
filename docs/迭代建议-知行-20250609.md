  **方案 A：紧急轮转（推荐，本周）**
  ```python
  # 1. 创建 src/gateway/rotating_audit.py
  import gzip, os, shutil
  from datetime import datetime
  from pathlib import Path
  
  class RotatingAuditLogger:
      """轮转审计日志：按大小 100MB / 按天滚动"""
      def __init__(self, log_file="data/audit.jsonl", max_bytes=100*1024*1024, backup_count=10):
          self._log_file = log_file
          self._max_bytes = max_bytes
          self._backup_count = backup_count
      
      def should_rotate(self) -> bool:
          return os.path.getsize(self._log_file) > self._max_bytes
      
      def rotate(self):
          """轮转：当前文件 → audit-YYYY-MM-DD-HH.jsonl.gz"""
          if not os.path.exists(self._log_file):
              return
          ts = datetime.now().strftime("%Y%m%d-%H%M%S")
          archive = f"data/audit-archive/audit-{ts}.jsonl.gz"
          os.makedirs("data/audit-archive", exist_ok=True)
          with open(self._log_file, 'rb') as f_in:
              with gzip.open(archive, 'wb') as f_out:
                  shutil.copyfileobj(f_in, f_out)
          # 清空原文件（保留链尾 prev_hash）
          open(self._log_file, 'w').close()
  ```
  
  **方案 B：清理历史（紧急止血，1h 完成）**
  ```bash
  # 保留最近 30 天，压缩 30 天前的
  python3 -c "
  import json, gzip, os
  from datetime import datetime, timedelta
  cutoff = (datetime.now() - timedelta(days=30)).timestamp()
  keep, archive = [], []
  with open('data/audit.jsonl') as f:
      for line in f:
          try:
              d = json.loads(line)
              (keep if d.get('timestamp', 0) >= cutoff else archive).append(line)
          except: pass
  with open('data/audit.jsonl', 'w') as f:
      f.writelines(keep)
  with gzip.open('data/audit-archive-20250609.jsonl.gz', 'wt') as f:
      f.writelines(archive)
  print(f'Kept {len(keep)} records, archived {len(archive)} records')
  "
  ```
